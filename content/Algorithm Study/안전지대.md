def solution(board):
    answer = 0
    dx = [-1, 0, 1]
    dy = [-1, 0, 1]
    
    for x in range (len(board)):
        for y in range (len(board[0])):
            count = 0
            for dir_x in dx:
                for dir_y in dy:
                    temp_x = x+dir_x
                    temp_y = y+dir_y
                    if temp_x >= 0 and temp_x < len(board) and temp_y >= 0 and temp_y < len(board):
                        if board[temp_x][temp_y] == 1:
                            count = count + 1
            if count == 0 :
                answer = answer + 1
                
    return answer
